import java.util.Arrays;
import java.util.Scanner;

public class Main {
    public static void main(String[] args) {

        System.out.println("숫자 6개 입력해주세요(※숫자 간격 쉼표 구분");
        System.out.println("가장 큰 수는 : "+ Arrays.stream(new Scanner(System.in).nextLine().trim().split(", ")).mapToInt(Integer::parseInt).max().getAsInt());
    }
}
